local base = require('scene_base')
local Scene = class('Step', base)

-- 异步加载生命周期 开始加载
function Scene:onLoadStart(a,b,c)
    log.print("开始加载>>",a,b,c)
end
-- 异步加载生命周期 加载过程
function Scene:onLoading(ao,progress,a,b,c)
    log.print("正在加载>>",progress,a,b,c);
end
-- 异步加载生命周期 加载完毕
function Scene:onLoadEnd(ao,a,b,c)
    log.print("加载完毕>>",a,b,c);
end
-- 加载完毕 适合直接加载的场景 只携带场景参数
function Scene:onLoad(a,b,c)
    log.print("进入场景>>",a,b,c)
    scene:load('Game',1,2,3)
end

return Scene
